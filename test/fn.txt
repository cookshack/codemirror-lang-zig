fn one();

==>

# One fn proto w arg

fn one(a : u32);

==>
Program(Decl(FnProto(Identifier,ParamDeclList(ParamDecl(Identifier, ParamType(TypeExpr(PrimaryTypeExpr(Identifier))))))))
