# init

const p = Point{
    .x = 0.1,
};

==>
Program(Decl(VarDecl(const,
                     Name,
                     Expr(PrimaryExpr(TypeExpr(Identifier),
                                      InitList(FieldInit(Identifier,
                                                         Expr(PrimaryExpr(TypeExpr(Float))))))))))

# dot

const a: Point = .{
  .x = 5,
};

==>
Program(Decl(VarDecl(const,
                     Name,
                     TypeExpr(Identifier),
                     Expr(PrimaryExpr(TypeExpr(InitList(FieldInit(Identifier,
                                                                  Expr(PrimaryExpr(TypeExpr(Integer)))))))))))

# dangle

// both for...| and Point are curlySuffixExpr which ends in an optional InitList
const p = for (a) |x| Point {};

==>
Program(LineComment,
        Decl(VarDecl(const,
                     Name,
                     Expr(PrimaryExpr(LoopExpr(ForExpr(ForPrefix(for,
                                                                 ForArgList(ForItem(Expr(PrimaryExpr(TypeExpr(Identifier))))),
                                                                 PtrListPayload(Identifier)),
                                                       Expr(PrimaryExpr(TypeExpr(Identifier),
                                                                        InitList)))))))))

# decl

const Point = packed struct (u32) {
  //! container doc comment

  // comment
  x: u16,
  /// y doc comment
  y: u16 align(16) = 3
};

==>
Program(Decl(VarDecl(const,
                     Name,
                     Expr(PrimaryExpr(TypeExpr(ContainerDecl(packed,
                                                             ContainerDeclAuto(ContainerDeclType(struct,
                                                                                                 Expr(PrimaryExpr(TypeExpr(Identifier)))),
                                                                               ContainerDocComment,
                                                                               LineComment,
                                                                               ContainerField(Identifier,
                                                                                              TypeExpr(Identifier)),
                                                                               ContainerField(DocComment,
                                                                                              Identifier,
                                                                                              TypeExpr(Identifier),
                                                                                              ByteAlign(align,
                                                                                                        Expr(PrimaryExpr(TypeExpr(Integer)))),
                                                                                              Expr(PrimaryExpr(TypeExpr(Integer))))))))))))
