# Top level const

const a;

==>
Program(Decl(VarDecl(const, Name)))

# Top level vars

const a;
var b;
threadlocal var c : u32;
const d : string;

==>
Program(Decl(VarDecl(const, Name)),
        Decl(VarDecl(var, Name)),
        Decl(threadlocal, VarDecl(var, Name, TypeExpr(Identifier))),
        Decl(VarDecl(const, Name, TypeExpr(Identifier))))

# Type

const a : u32;

==>
Program(Decl(VarDecl(const,
                     Name,
                     TypeExpr(Identifier))))

# Type and val

const a : u32 = 0;

==>
Program(Decl(VarDecl(const,
                     Name,
                     TypeExpr(Identifier),
                     Expr(TypeExpr(Integer)))))

# export

export var a : u32 = 0;

==>
Program(Decl(export,
             VarDecl(var,
                     Name,
                     TypeExpr(Identifier),
                     Expr(TypeExpr(Integer)))))

# extern

extern "c" var x : u32;

==>
Program(Decl(extern,
             StringLiteralSingle,
             VarDecl(var,
                     Name,
                     TypeExpr(Identifier))))
