# Top level const

const a;

==>
Program(Decl(VarDecl(Const, Name)))

# Top level vars

const a;
var b;
threadlocal var c : u32;
const d : string;

==>
Program(Decl(VarDecl(Const, Name)),
        Decl(VarDecl(Var, Name)),
        Decl(threadlocal, VarDecl(Var, Name, TypeExpr(Identifier))),
        Decl(VarDecl(Const, Name, TypeExpr(Identifier))))

# Type

const a : u32;

==>
Program(Decl(VarDecl(Const,
                     Name,
                     TypeExpr(Identifier))))

# Type and val

const a : u32 = 0;

==>
Program(Decl(VarDecl(Const,
                     Name,
                     TypeExpr(Identifier),
                     Expr(PrimaryExpr(TypeExpr(Integer))))))

# export

export var a : u32 = 0;

==>
Program(Decl(export,
             VarDecl(Var,
                     Name,
                     TypeExpr(Identifier),
                     Expr(PrimaryExpr(TypeExpr(Integer))))))

# extern

extern "c" var x : u32;

==>
Program(Decl(Extern,
             StringLiteralSingle,
             VarDecl(Var,
                     Name,
                     TypeExpr(Identifier))))
