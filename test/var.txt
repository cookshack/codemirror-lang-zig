# Top level const

const a;

==>
Program(Decl(VarDecl(Const, Name)))

# Top level vars

const a;
var b;
var c : u32;
const d : string;

==>
Program(Decl(VarDecl(Const, Name)),
        Decl(VarDecl(Var, Name)),
        Decl(VarDecl(Var, Name, TypeExpr(Identifier))),
        Decl(VarDecl(Const, Name, TypeExpr(Identifier))))

# Type

const a : u32;

==>
Program(Decl(VarDecl(Const,
                     Name,
                     TypeExpr(Identifier))))

# Type and val

const a : u32 = 0;

==>
Program(Decl(VarDecl(Const,
                     Name,
                     TypeExpr(Identifier),
                     Expr(PrimaryExpr(CurlySuffixExpr(TypeExpr(Integer)))))))
