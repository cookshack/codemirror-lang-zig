# shortcut

const e = error.OutOfPie;

==>
Program(Decl(GlobalVarDecl(VarDeclProto(const,
                                        Name),
                           Expr(TypeExpr(error,
                                         Identifier)))))

# set decl

const e : error{ OutOfPie, OutOfJuice };

==>
Program(Decl(GlobalVarDecl(VarDeclProto(const,
                                        Name,
                                        TypeExpr(ErrorSetDecl(error,
                                                              Identifier,
                                                              Identifier))))))

# union type

const pies : anyerror!u32 = error.OutOfPie;

==>
Program(Decl(GlobalVarDecl(VarDeclProto(const,
                                        Name,
                                        TypeExpr(ErrorUnionExpr(Identifier,
                                                                TypeExpr(Identifier)))),
                           Expr(TypeExpr(error,
                                         Identifier)))))
