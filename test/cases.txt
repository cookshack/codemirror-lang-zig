# Empty doc

==>
Program()

# Top level const

const a;

==>
Program(Decl(VarDecl(Const, Name)))

# Top level vars

const a;
var b;
var c : u32;
const d : string;

==>
Program(Decl(VarDecl(Const, Name)),
        Decl(VarDecl(Var, Name)),
        Decl(VarDecl(Var, Name, TypeExpr(PrimaryTypeExpr(Identifier)))),
        Decl(VarDecl(Const, Name, TypeExpr(PrimaryTypeExpr(Identifier)))))

# Block statement

fn one() {
  return;
}

==>
Program(Decl(FnProto(Identifier,ParamDeclList()),
             Block(Statement(Return))))

# Assign in block

fn one() {
  a = 1;
}

==>
Program(Decl(FnProto(Identifier,ParamDeclList()),
             Block(Statement(CurlySuffixExpr(TypeExpr(PrimaryTypeExpr(Identifier))),
                             AssignOp,
                             Expr(PrimaryExpr(CurlySuffixExpr(TypeExpr(PrimaryTypeExpr(Integer)))))))))

# import

const std = @import("std");

==>
Program
(Decl
 (VarDecl
  (Const,
   Name,
   Expr(PrimaryExpr(CurlySuffixExpr(TypeExpr(PrimaryTypeExpr(BuiltinIdentifier,
                                                             FnCallArgs(Expr(PrimaryExpr(CurlySuffixExpr(TypeExpr(PrimaryTypeExpr(StringLiteral))))))))))))))
