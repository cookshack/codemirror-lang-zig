# plain

test  {
}

==>
Program(TestDecl(test,Block))

# named

test "check" {
}

==>
Program(TestDecl(test,TestName,Block))

# named with try

test "check" {
  try check();
}

==>
Program(TestDecl(test,
                 TestName,
                 Block(Statement(PrefixExpr(PrefixOp(try),
                                            TypeExpr(Identifier,
                                                     FnCallArgs))))))
